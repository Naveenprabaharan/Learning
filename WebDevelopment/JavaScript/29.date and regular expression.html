            //Basic Regular expression
            var RegEx = /[0-9]/;
            console.log(RegEx.test('some text'))//false
            console.log(RegEx.test('some 12345 text'))//True

            var RegEx = new RegExp('[0-9]');
            console.log(RegEx.test('some text'))//false
            console.log(RegEx.test('some 12345 text'))//True
