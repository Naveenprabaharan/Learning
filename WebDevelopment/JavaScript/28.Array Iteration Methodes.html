            //4.every => every element need to pass some Condition
            //eg:4=>chack student got passed or not Based on array element

            function getResult(marklist)
            {
                var result = marklist.every(function(mark)
                {
                    return mark >= 45
                })
                if (result){
                    console.log('student passed in all subject!');
                }else{
                    console.log('it seems you failed in some Subjects!')
                }
            }

            getResult([45,67,87,65,90])
            getResult([34,56,23,87,22])

            //5.some => some element need to pass some Condition
            //eg:4=>chack student got passed some subject Based on array element

            function getResults(marklist)
            {
                var result = marklist.some(ispass)
                    if(result)
                    {
                        console.log('student is passed some subject!')
                    }else{
                        console.log('student is passed all subject!')
                    }
                
            };

            function ispass(mark){
                if(mark !== 'pass'){
                    return mark;
                }
            }

            getResults(['pass','fail','pass','fail'])
            getResults(['pass','pass','pass','pass'])
