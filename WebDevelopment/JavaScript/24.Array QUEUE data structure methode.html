Array QUEUE data structure methode

script>
            var meal = new Array('breakFast')
            //add item into queue using push function
            var count = meal.push('lunch','dinner')
            console.log(count)
            
            //display our queue
            console.log(meal)

            //remove first item using shift,removed object is return value
            var firstMeal = meal.shift()
            console.log(firstMeal)
            console.log(meal)

            //add item as first in queue
            meal.unshift('breakFast')
            console.log(meal)

            //remove item in last ,removed object is return value
            var lastMeal = meal.pop()
            console.log(lastMeal)
            console.log(meal)
