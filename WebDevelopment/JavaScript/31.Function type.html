            //to find factorial;

            function factorial(num)
            {
                if (num <=1)
                {
                    return num;
                }else{
                    return num * factorial(num-1);
                }
            }

            console.log('factoraial of 5',factorial(5))

            //copy factorial function to another function 
            var newFactorial = factorial
            //factorial = null
            console.log('factoraial of 5',newFactorial(5))//it will say error because in our original function 
            //we have call 'return num * "factorial"(num-1);' inthis factorial become null so now this function is not working

            //to avoide that type type of referene error now we are using "calle" methode 
            function factorial(num)
            {
                if (num <=1)
                {
                    return num;
                }else{
                    return num * arguments.callee(num-1);
                }
            }
            console.log('factoraial of 5',factorial(5))

            //copy factorial function to another function 
            var newFactorial = factorial
            factorial = null
            console.log('factoraial of 5',newFactorial(5))// reference error got eliminated
